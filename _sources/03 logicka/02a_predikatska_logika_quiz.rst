.. quizq::
                  
    .. mchoice:: valjane
        :answer_a: ¬ (∃ x)(P(x) ∧ ¬ P(x)))
        :answer_b: (∀ x)(¬ (P(x) ⇒ Q(x)) ⇔ P(x) ∧ ¬ Q(x)))
        :answer_c: (∃ x)(P(x) ⇒ (∀ x)(P(x)))
        :answer_d: ¬ (∃ x)(P(x)) ⇔ (∀ x)(¬ P(x))
        :correct: a,b,c,d
